001package org.anarres.typeserializer.core;
002
003import java.lang.reflect.GenericArrayType;
004import java.lang.reflect.ParameterizedType;
005import java.lang.reflect.Type;
006import java.lang.reflect.TypeVariable;
007import java.lang.reflect.WildcardType;
008import java.util.Arrays;
009import java.util.EnumSet;
010import java.util.Set;
011import javax.annotation.CheckForNull;
012import javax.annotation.Nonnull;
013import org.anarres.typeserializer.core.impl.Utils;
014
015/**
016 *
017 * @author shevek
018 */
019public class GenericNameTypeVisitor extends TypeVisitor<StringBuilder, Void, RuntimeException> {
020
021    public static final GenericNameTypeVisitor INSTANCE = new GenericNameTypeVisitor();
022    private static final Type[] OBJECT_TYPE_ARRAY = new Type[]{Object.class};
023
024    public static enum Feature {
025
026        VarargsTopLevel, UnqualifiedImports, UnqualifiedJavaLangImports;
027    }
028    private final Set<Feature> features;
029
030    public GenericNameTypeVisitor(@Nonnull Feature... features) {
031        this.features = EnumSet.noneOf(Feature.class);
032        this.features.addAll(Arrays.asList(features));
033    }
034
035    private boolean isVarargs() {
036        return getDepth() == 1 && features.contains(Feature.VarargsTopLevel);
037    }
038
039    @Override
040    public Void visitParameterizedType(ParameterizedType type, StringBuilder out) {
041        visit(type.getRawType(), out);
042        Type[] typeArguments = type.getActualTypeArguments();
043        if (Utils.isNotEmpty(typeArguments)) {
044            out.append("<");
045            boolean b = false;
046            for (Type ptype : typeArguments) {
047                if (b)
048                    out.append(", ");
049                else
050                    b = true;
051                visit(ptype, out);
052            }
053            out.append(">");
054        }
055        return null;
056    }
057
058    @Override
059    public Void visitGenericArrayType(GenericArrayType type, StringBuilder out) {
060        visit(type.getGenericComponentType(), out);
061        if (isVarargs())
062            out.append("...");
063        else
064            out.append("[]");
065        return null;
066    }
067
068    private void appendBounds(@Nonnull StringBuilder out, @CheckForNull Type[] bounds, @Nonnull String keyword) {
069        if (Utils.isNotEmpty(bounds)) {
070            out.append(keyword);
071            boolean b = false;
072            for (Type bound : bounds) {
073                if (b)
074                    out.append(" & ");
075                else
076                    b = true;
077                visit(bound, out);
078            }
079        }
080    }
081
082    @Override
083    public Void visitTypeVariable(TypeVariable<?> type, StringBuilder out) {
084        out.append(type.getName());
085        appendBounds(out, type.getBounds(), " extends ");
086        return null;
087    }
088
089    @Override
090    public Void visitWildcardType(WildcardType type, StringBuilder out) {
091        out.append("?");
092        Type[] upperBounds = type.getUpperBounds();
093        if (!Arrays.equals(upperBounds, OBJECT_TYPE_ARRAY))
094            appendBounds(out, upperBounds, " extends ");
095        appendBounds(out, type.getLowerBounds(), " super ");
096        return null;
097    }
098
099    protected void visitClassName(@Nonnull Class<?> type, StringBuilder out) {
100        if (features.contains(Feature.UnqualifiedImports))
101            out.append(type.getSimpleName());
102        else if (features.contains(Feature.UnqualifiedJavaLangImports) && Utils.equals("java.lang", Utils.getPackageName(type)))
103            out.append(type.getSimpleName());
104        else
105            out.append(type.getName());
106    }
107
108    @Override
109    public Void visitClass(Class<?> type, StringBuilder out) {
110        int array = 0;
111        while (type.isArray()) {
112            type = type.getComponentType();
113            array++;
114        }
115        visitClassName(type, out);
116        boolean varargs = array > 0 && isVarargs();
117        if (varargs)
118            array--;
119        for (int i = 0; i < array; i++)
120            out.append("[]");
121        if (varargs)
122            out.append("...");
123        return null;
124    }
125
126}
