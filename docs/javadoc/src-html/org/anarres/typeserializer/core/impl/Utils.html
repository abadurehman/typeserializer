001package org.anarres.typeserializer.core.impl;
002
003import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
004import java.lang.reflect.Type;
005import java.util.Collections;
006import java.util.HashMap;
007import java.util.Map;
008import javax.annotation.CheckForNull;
009import javax.annotation.Nonnull;
010
011/**
012 *
013 * @author shevek
014 */
015public class Utils {
016
017    public static final Type[] EMPTY_TYPE_ARRAY = new Type[0];
018
019    /**
020     * The package separator character: <code>'&#x2e;' == {@value}</code>.
021     */
022    public static final char PACKAGE_SEPARATOR_CHAR = '.';
023
024    /**
025     * The inner class separator character: <code>'$' == {@value}</code>.
026     */
027    public static final char INNER_CLASS_SEPARATOR_CHAR = '$';
028
029    public static boolean isNotEmpty(@CheckForNull Object[] a) {
030        if (a == null)
031            return false;
032        return a.length > 0;
033    }
034
035    public static boolean equals(@CheckForNull Object a, @CheckForNull Object b) {
036        return (a == b) || (a != null && a.equals(b));
037    }
038
039    public static int hashCode(@CheckForNull Object o) {
040        return o != null ? o.hashCode() : 0;
041    }
042
043    /**
044     * Maps a primitive class name to its corresponding abbreviation used in array class names.
045     */
046    private static final Map<String, String> abbreviationMap;
047
048    /**
049     * Feed abbreviation maps
050     */
051    static {
052        final Map<String, String> m = new HashMap<String, String>();
053        m.put("int", "I");
054        m.put("boolean", "Z");
055        m.put("float", "F");
056        m.put("long", "J");
057        m.put("short", "S");
058        m.put("byte", "B");
059        m.put("double", "D");
060        m.put("char", "C");
061        m.put("void", "V");
062        abbreviationMap = Collections.unmodifiableMap(m);
063    }
064
065    @Nonnull
066    public static String getPackageName(@Nonnull Class<?> c) {
067        String className = c.getName();
068        int i = className.lastIndexOf(PACKAGE_SEPARATOR_CHAR);
069        if (i == -1)
070            return "";
071        return className.substring(0, i);
072    }
073
074    // ----------------------------------------------------------------------
075    /**
076     * Converts a class name to a JLS style class name.
077     *
078     * @param className  the class name
079     * @return the converted name
080     */
081    @Nonnull
082    private static String toCanonicalName(@Nonnull String className) {
083        if (className.endsWith("[]")) {
084            final StringBuilder classNameBuffer = new StringBuilder();
085            while (className.endsWith("[]")) {
086                className = className.substring(0, className.length() - 2);
087                classNameBuffer.append("[");
088            }
089            final String abbreviation = abbreviationMap.get(className);
090            if (abbreviation != null) {
091                classNameBuffer.append(abbreviation);
092            } else {
093                classNameBuffer.append("L").append(className).append(";");
094            }
095            className = classNameBuffer.toString();
096        }
097        return className;
098    }
099
100    // Class loading
101    // ----------------------------------------------------------------------
102    /**
103     * Returns the class represented by {@code className} using the
104     * {@code classLoader}.  This implementation supports the syntaxes
105     * "{@code java.util.Map.Entry[]}", "{@code java.util.Map$Entry[]}",
106     * "{@code [Ljava.util.Map.Entry;}", and "{@code [Ljava.util.Map$Entry;}".
107     *
108     * @param classLoader  the class loader to use to load the class
109     * @param className  the class name
110     * @param initialize  whether the class must be initialized
111     * @return the class represented by {@code className} using the {@code classLoader}
112     * @throws ClassNotFoundException if the class is not found
113     */
114    public static Class<?> getClass(
115            final ClassLoader classLoader, final String className, final boolean initialize) throws ClassNotFoundException {
116        try {
117            Class<?> clazz;
118            if (abbreviationMap.containsKey(className)) {
119                final String clsName = "[" + abbreviationMap.get(className);
120                clazz = Class.forName(clsName, initialize, classLoader).getComponentType();
121            } else {
122                clazz = Class.forName(toCanonicalName(className), initialize, classLoader);
123            }
124            return clazz;
125        } catch (final ClassNotFoundException ex) {
126            // allow path separators (.) as inner class name separators
127            final int lastDotIndex = className.lastIndexOf(PACKAGE_SEPARATOR_CHAR);
128
129            if (lastDotIndex != -1) {
130                try {
131                    return getClass(classLoader, className.substring(0, lastDotIndex)
132                            + INNER_CLASS_SEPARATOR_CHAR + className.substring(lastDotIndex + 1),
133                            initialize);
134                } catch (final ClassNotFoundException ex2) { // NOPMD
135                    // ignore exception
136                }
137            }
138
139            throw ex;
140        }
141    }
142
143    @SuppressFBWarnings("NP_PARAMETER_MUST_BE_NONNULL_BUT_MARKED_AS_NULLABLE")
144    public static void assertNotNull(@CheckForNull Object object, @Nonnull String message) {
145        if (object == null)
146            throw new NullPointerException(message);
147    }
